let scene, camera, renderer; function init() { scene = new THREE.Scene(); scene.background = new THREE.Color(0xdddddd); camera = new THREE.PerspectiveCamera(40,window.innerWidth/window.innerHeight,1,5000); camera.rotation.y = 45/180*Math.PI; camera.position.x = -4; camera.position.y = 2; camera.position.z = -5; controls = new THREE.OrbitControls(camera); controls.addEventListener('change', renderer); var hemiLight = new THREE.HemisphereLight( 0xffffff, 0x444444 ); hemiLight.position.set( 0, 20, 0 ); scene.add( hemiLight ); var dirLight = new THREE.DirectionalLight( 0xffffff ); dirLight.position.set( - 3, 10, - 10 ); dirLight.castShadow = true; dirLight.shadow.camera.top = 2; dirLight.shadow.camera.bottom = - 2; dirLight.shadow.camera.left = - 2; dirLight.shadow.camera.right = 2; dirLight.shadow.camera.near = 0.1; dirLight.shadow.camera.far = 40; scene.add( dirLight ); renderer = new THREE.WebGLRenderer({antialias:true}); renderer.setSize(window.innerWidth,window.innerHeight); renderer.gammaOutput = true; renderer.gammaFactor = 2.2; document.body.appendChild(renderer.domElement); let loader = new THREE.GLTFLoader(); loader.load('https://threejs.org/examples/models/gltf/Soldier.glb', function(gltf){ car = gltf.scene.children[0]; scene.add(gltf.scene); animate(); }); } function animate() { renderer.render(scene,camera); requestAnimationFrame(animate); } init();
